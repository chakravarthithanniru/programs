import java.util.Scanner;
class A
{
  public static void main(String[] args)
  {
     Scanner obj=new Scanner(System.in);
     int a=obj.nextInt();
     int b=obj.nextInt();
     int c=0;
     for(int i=a;i<=b;i++)
     {
        if(i%2==0)
        {
          c++;
         }
     }
     System.out.println(c);
   }
}
